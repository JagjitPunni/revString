                    //Reverse the string
                    //Logic function
                    function reverseString(userString){

                        let revString = [];

                        
                        //reverse a string using a for loop
                        for (let i = userString.length - 1; i >= 0; i--) {
                            revString += userString[i];
                            
                        }
                        return revString;
                    }


                    //Display the reversed string to the user
                    //View function
                    function displayString(revString,userString){

                        //write to the page
                        document.getElementById('msg').innerHTML = `${revString} is the reverse of ${userString}.`;
                        //show the alert box
                        document.getElementById('alert').classList.remove('invisible');
                        
                    }

Overview

This JavaScript code uses functions, DOM manipulation, and loops to reverse a user input string and display it on an HTML page. The code is divided into three functions:

  • getValue() - Retrieves user input, calls reverse function and passes the result to display function.
  • reverseString(userString) - Reverses a string using a for loop and returns the result.
  • displayString(revString,userString) - Displays the reversed and original strings to the user.
